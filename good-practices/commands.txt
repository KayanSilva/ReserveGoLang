Golangci-cli
Para executar no Linux
docker run -t --rm -v $(pwd):/app -w /app golangci/golangci-lint:v2.6.1 golangci-lint run
Para executar no Windows
docker run -t --rm -v "$(Get-Location):/app" -w /app golangci/golangci-lint:v2.6.1 golangci-lint run
